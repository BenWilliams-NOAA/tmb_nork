           rtmb$fish_age_pred) %>% 
  pivot_longer(-age) %>% 
  mutate(year = rep(fish_age_yrs, length(ages)),
         groups = 'pred2') %>% 
  bind_rows(fac) -> df2

  ggplot(aes(age, value, color = groups)) + 
  geom_line() +
  facet_wrap(~year)

Code 
df2 %>% 
  select(age, value, year, groups) %>% 
  filter(groups!='obs') %>% 
  pivot_wider(names_from=groups, values_from = value) %>%
